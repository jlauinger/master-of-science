%% Configuration for the styling of research questions
%% =====================================================================================================================

%% Boxes for RQ answers and insights -----------------------------------------------------------------------------------

\usepackage{morewrites}
\usepackage{tcolorbox}

%% RQ answer  box formatting -------------------------------------------------------------------------------------------

\newenvironment{answerToRQ}[1][\todo{specify RQ!}]{
    \begin{tcolorbox}[boxsep=3pt, enlarge top by=5pt, arc=0pt, title=Answer to~#1]
}{\end{tcolorbox}}

%% Insight box formatting ----------------------------------------------------------------------------------------------

\newcounter{insightNumber}

\newenvironment{insight}{%
    \refstepcounter{insightNumber}
    \begin{tcolorbox}[boxsep=3pt, enlarge top by=5pt, arc=0pt, title=Insight~\theinsightNumber]
}{\end{tcolorbox}}

%% General hero box ----------------------------------------------------------------------------------------------------

\newenvironment{hero}[1][\todo{provide title!}]{%
    \begin{tcolorbox}[boxsep=3pt, enlarge top by=5pt, arc=0pt, title=#1]
}{\end{tcolorbox}}
