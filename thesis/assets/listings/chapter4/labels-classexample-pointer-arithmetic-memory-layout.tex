func (p *page) branchPageElement(index uint16) *branchPageElement {
    off := uintptr(index) * unsafe.Sizeof(branchPageElement{})
    return (*branchPageElement)(unsafe.Pointer(uintptr(unsafe.Pointer(p)) + unsafe.Sizeof(*p) + off))
}
