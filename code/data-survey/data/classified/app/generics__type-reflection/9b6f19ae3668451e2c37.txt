// scalarToHeader creates a Header from a scalar value
func scalarToHeader(a interface{}) *storage.Header {
	hdr := borrowHeader()
	hdr.Ptr = getPointer(a)
	hdr.L = 1
	hdr.C = 1
	return hdr
}
